    }

  checkAnswersSelectedForRange(value: Array<any>, keyHeader: number, keyFooter: number): any {
    for (let index = keyHeader; index <= keyFooter; index++) {
      if (value[index - 1] && value[index - 1].answer) {
        return true;
      }
    }
    return false;
  }

    let flag = false;

        if (i !== key && i !== value.length - 1) {
          if (item.answer) {
            flag = true;
          }
        }
      });

      value.forEach((item, i) => {
        if (i !== key) {

            flag = true;
          }
        }
      });

  }
